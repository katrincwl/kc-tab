kc-tab

jQuery plugin: Make dom element into tabs easily

HTML DOM example

<div class="container" >
  <!-- Tab buttons -->
  <div class="kc_tab_btn" data-tabbox-id="tabbox_1" ></div>
  <div class="kc_tab_btn" data-tabbox-id="tabbox_2" ></div>
  <!-- Tab contents -->
  <div class="kc_tab_content" data-tabbox-id="tabbox_1" > {tab 1 content} </div>
  <div class="kc_tab_content" data-tabbox-id="tabbox_2" > {tab 2 content} </div>
</div>

Javascript Example

$('.container').kc_tab({
    default_tab : 0,
    tab_btn_cls : 'kc_tab_btn',
    tab_content_cls : 'kc_tab_content',
    active_tab_cls : 'kc_active_tab',
    active_content_cls : 'kc_active_content'
});

CSS Example

.kc_tab_content{display:none;}
.kc_active_tab{background-color:#10334B;}
.kc_active_content{display:block;}
